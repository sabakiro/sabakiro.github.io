模式匹配

2023-08-05 843 words
rust

match和if let

match

语法

match target {
    模式1 => 表达式1,
    模式2 => {
        语句1;
        语句2;
        表达式2
    },
    _ => 表达式3
}

示例


enum Direction {
    East,
    West,
    North,
    South,
}

fn main() {
    let dire = Direction::South;
    match dire {
        Direction::East => println!("East"),
        //单分支，多模式
        Direction::North | Direction::South => {
            println!("South or North");
        },
        _ => println!("West"),
    };
}
  • match 的匹配必须要穷举出所有可能，因此这里用 _ 来代表未列出的所有可能性，相当于switch中的default
  • match 的每一个分支都必须是一个表达式，且所有分支的表达式最终返回值的类型必须相同
  • X | Y，类似逻辑运算符 或，代表该分支可以匹配 X 也可以匹配 Y，只要满足一个即可

if let匹配简化match

let v:Option<u8>=Some(3);
match v{
    Some(3)=>println!("three"),
    _=>(),
}
//使用if let简化
let v:Option<u8>=Some(3);
if let Some(3)=v{
    println!("three");
}

matches!宏

可以将一个表达式跟模式进行匹配，然后返回匹配的结果 true or false

#[derive(Debug)]
enum MyEnum {
    Foo,
    Bar,
}

fn main() {
    let v = vec![MyEnum::Foo, MyEnum::Bar, MyEnum::Foo];
    //error,无法将 x 直接跟一个枚举成员进行比较
    // v.iter().filter(|x| x == MyEnum::Foo);
    
    //使用match匹配
    let v=v.iter().filter(|x| match x {
        MyEnum::Foo => true,
        _ => false,
    });
    for e in v  {
        println!("{:?}",e);
    }
    
    //使用matches！(变量，模式)
    let v=v.iter().filter(|x| matches!(x,MyEnum::Foo));
    for e in v  {
        println!("{:?}",e);
    }
}

模式

while let

// Vec是动态数组
let mut stack = Vec::new();

// 向数组尾部插入元素
stack.push(1);
stack.push(2);
stack.push(3);

// stack.pop从数组尾部弹出元素
while let Some(top) = stack.pop() {
    println!("{}", top);
}

for

let v = vec!['a', 'b', 'c'];
//获取下标，元素值
for (index, value) in v.iter().enumerate() {
    println!("{} is at index {}", value, index);
}

不可驳模式

类似 let , for和match 都必须要求完全覆盖匹配

let Some(x) = some_option_value;

可驳模式

因为 if let和while let 允许匹配一种模式，而忽略其余的模式( 可驳模式匹配 )。

if let Some(x) = some_option_value {
    println!("{}", x);
}

通过序列匹配

let x = 5;

match x {
    1..=5 => println!("one through five"),
    _ => println!("something else"),
}

匹配守卫

匹配守卫（match guard）是一个位于 match 分支模式之后的额外 if 条件，

它能为分支模式提供更进一步的匹配条件。

let num = Some(4);
match num {
    Some(x) if x < 5 => println!("less than five: {}", x),
    Some(x) => println!("{}", x),
    None => (),
}

@绑定

#[derive(Debug)]
struct Point {
    x: i32,
    y: i32,
}
fn main() {
    enum Message {
        Hello { id: i32 },
    }
    let msg = Message::Hello { id: 5 };
    match msg {
        Message::Hello { id: id_variable @ 3..=7 } => {
            println!("Found an id in range: {}", id_variable)
        },
        _=>()
    }
    
    // 绑定新变量 p，同时对 Point 进行解构
    let p @ Point {x: px, y: py } = Point {x: 10, y: 23};
    
    let point = Point {x: 10, y: 5};
    if let p @ Point {x: 10, y} = point {
        println!("x is 10 and y is {} in {:?}", y, p);
    } else {
        println!("x was not 10 :(");
    }//x is 10 and y is 5 in Point { x: 10, y: 5 }
    
   //1.53后新增
	match 1 {
        num @ (1 | 2) => {
            println!("{}", num);
        }
        _ => {}
    }
}
