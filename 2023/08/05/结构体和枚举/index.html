结构体和枚举

2023-08-05 282 words
rust

结构体

struct User {
    active: bool,
    username: String,
}
fn main() {
    let user1 = User {
        active: true,
        username: String::from("someusername123"),
    };
    let user2 = User {
        username: String::from("ciallo"),
        ..user1
    };
    assert_eq!("ciallo", &user2.username);
    assert_eq!(true, user2.active);
}

所有权

#[derive(Debug)]
struct User {
    active: bool,
    username: String,
}
fn main() {
    let user1 = User {
        active: true,
        username: String::from("someusername123"),
    };
    let user2=User{
        active:false,
        username:user1.username,
    };
    println!("{:?}",user1);//error，被部分借用
    println!("{}",user1.username);//error, 被借用
    println!("{}",user1.active);
}

枚举

enum Stu {
    Age(i32),
    Name(String),
    TTK,
    
}
fn main() {
    let age=Stu::Age(10);
    let name=Stu::Name("ciallo world".to_string());
    let ttk=Stu::TTK;
    let print_stu=|e|{
        match e {
            Stu::Age(i)=>println!("age:{}",i),
            Stu::Name(name)=>println!("name:{}",name),
            Stu::TTK=>println!("TTK yizimo yizimo TTK"),
        }
    };
    print_stu(age);
    print_stu(name);
    print_stu(ttk);
}
