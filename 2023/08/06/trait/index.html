trait基础

//定义trait
pub trait Summary {
    //定义trait方法
    fn summarize(&self);
    //默认实现
    fn fuction(&self) {
        println!("summary function");
    }
}
pub struct A;
//实现trait
impl Summary for A {
    fn summarize(&self) {
        println!("impl trait Summary for A")
    }
}
fn function(x: &impl Summary) {
    x.summarize();
}
pub struct B;
impl Summary for B {
    fn summarize(&self) {}
    //重写trait的默认实现
    fn fuction(&self) {
        println!("B fuction")
    }
}
//返回值是实现trait的对象
fn C(x: impl Summary) -> impl Summary {
    x
}
fn main() {
    let a = A;
    function(&a);
    a.fuction();
    let b = B;
    b.fuction();
    let _ = C(b);
    /*
    impl trait Summary for a
    summary function
    B function
     */
}

trait约束

pub trait Summary {}
pub struct A;
impl Summary for A {}
fn function<T:Summary>(_:&T){}
fn main() {
    let a=A;
    function(&a);
}

多重约束

pub trait Summary {}
#[derive(Clone, Copy)]
pub struct A;
impl Summary for A {}
fn function<T:Summary+Copy>(_:&T){}
fn main() {
    let a=A;
    function(&a);
}

where约束

pub trait Summary {}
#[derive(Clone, Copy)]
pub struct A;
impl Summary for A {}
fn function<T>(_: &T)
where
    T: Summary + Copy,
{}
fn main() {
    let a = A;
    function(&a);
}

trait对象

Box实现trait对象

trait Draw {
    fn draw(&self) -> String;
}

impl Draw for u8 {
    fn draw(&self) -> String {
        format!("u8: {}", *self)
    }
}

impl Draw for f64 {
    fn draw(&self) -> String {
        format!("f64: {}", *self)
    }
}

// 若 T 实现了 Draw 特征， 则调用该函数时传入的 Box<T> 可以被隐式转换成函数参数签名中的 Box<dyn Draw>
fn draw1(x: Box<dyn Draw>) {
    // 由于实现了 Deref 特征，Box 智能指针会自动解引用为它所包裹的值，然后调用该值对应的类型上定义的 `draw` 方法
    x.draw();
}
fn main() {
    let x = 1.1f64;
    let y = 8u8;
    // x 和 y 的类型 T 都实现了 `Draw` 特征，因为 Box<T> 可以在函数调用时隐式地被转换为特征对象 Box<dyn Draw> 
    // 基于 x 的值创建一个 Box<f64> 类型的智能指针，指针指向的数据被放置在了堆上
    draw1(Box::new(x));
    // 基于 y 的值创建一个 Box<u8> 类型的智能指针
    draw1(Box::new(y));
}

& 实现trait对象

trait Draw {
    fn draw(&self) -> String;
}
impl Draw for u8 {
    fn draw(&self) -> String {
        format!("u8: {}", *self)
    }
}
impl Draw for f64 {
    fn draw(&self) -> String {
        format!("f64: {}", *self)
    }
}
fn draw2(x: &dyn Draw) {
    x.draw();
}
fn main() {
    let x = 1.1f64;
    let y = 8u8;
    draw2(&x);
    draw2(&y);
}

静态分发

泛型是在编译期完成处理的：编译器会为每一个泛型参数对应的具体类型生成一份代码，这种方式是**静态分发(static dispatch)**，因为是在编译期完成的，对于运行期性能完全没有任何影响。

动态分发

直到运行时，才能确定需要调用什么方法。

trait对象的限制

  • 方法的返回类型不能是 Self
  • 方法没有任何泛型参数

关联类型

//定义关联类型
pub trait my_iterator {
    //type Item;
    type Item:Copy;//添加trait约束
    fn next(&mut self)->Option<Self::Item>;
}
pub struct Vector{
  storage:Vec<i32>,
}
//实现关联类型
impl my_iterator for Vector {
    type Item = i32;
    fn next(&mut self)->Option<Self::Item> {
        self.storage.pop()
    }
}

默认泛型类型参数

pub trait Test<T=i32> {
    type Item;
    fn test(&self,other:T)->Self::Item ;
}
pub struct A;
impl Test for A {
  type Item = i32;
    fn test(&self,other:i32)->Self::Item {
        10
    }
}
pub struct B;
impl Test<u8> for B {
    type Item = i32;
    fn test(&self,other:u8)->Self::Item {
        10
    }
}

调用同名的方法

有&self参数的情况下

trait Pilot {
    fn fly(&self);
}
struct Human;
impl Pilot for Human {
    fn fly(&self) {
        println!("This is your captain speaking.");
    }
}
impl Human {
    fn fly(&self) {
        println!("*waving arms furiously*");
    }
}
fn main() {
    let person = Human;
    Pilot::fly(&person); // 调用Pilot特征上的方法
    person.fly(); // 调用Human类型自身的方法
}

没有&self参数的情况下

trait Animal {
  fn baby_name() -> String;
}
struct Dog;
impl Dog {
  fn baby_name() -> String {
      String::from("Spot")
  }
}
impl Animal for Dog {
  fn baby_name() -> String {
      String::from("puppy")
  }
}
fn main() {
  //调用自己的
  println!("A baby dog is called a {}", Dog::baby_name());
  //完全限定语法
  println!("A baby dog is called a {}", <Dog as Animal>::baby_name());
}

在外部类型上实现外部trait

trait限制

孤儿规则：就是特征或者类型必需至少有一个是本地的，才能在此类型上定义特征。

newtype 模式

作用：绕过孤儿规则的方法。

具体实现：就是为一个元组结构体创建新类型。该元组结构体封装有一个字段，该字段就是希望实现特征的具体类型。

use std::fmt;

struct Wrapper(Vec<String>);

impl fmt::Display for Wrapper {
    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
        write!(f, "[{}]", self.0.join(", "))
    }
}

fn main() {
    let w = Wrapper(vec![String::from("hello"), String::from("world")]);
    println!("w = {}", w);
}
