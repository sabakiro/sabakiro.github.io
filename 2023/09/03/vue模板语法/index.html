vue模板语法

2023-09-03 215 words
vue

文本插值

将数据解释为纯文本

<script setup>
import { ref } from 'vue'
const value = ref('<h1>ciallo world</h1>')

</script>

<template>
  <p>{{ value }}</p>
</template>

插入html

<script setup>
import { ref } from 'vue'
const value = ref('<h1>ciallo world</h1>')

</script>

<template>
<!--不能使用 v-html 来拼接组合模板-->
  <p v-html="value"></p>
</template>

绑定多个值

<script setup>
import { ref } from 'vue'
const value = ref(
  {
    id: '10086',
    class: 'ciallo',
  })
</script>

<template>
  <p v-bind="value">{{ value.id }}:{{ value.class }}</p>
  <div v-bind="value">
    <input v-model="value.id"><br />
    <input v-model="value.class">
  </div>
</template>
