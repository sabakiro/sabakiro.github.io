    vue类与样式绑定

    2023-09-03 760 words
    vue

    内联字面量的形式

    <script setup>
    import { ref } from 'vue'

    const isActive = ref(true)
    const isWidth = ref(true)
    </script>

    <template>
      <div class="height" :class="{ active: isActive, width: isWidth }"></div>
    </template>

    <style>
    .active {
      background-color: red;
    }

    .width {
      width: 500px;

    }

    .height {
      height: 60px;
    }
    </style>

    绑定对象

    <script setup>
      import { ref } from 'vue'

      const classObject = ref({
        active: true,
        width: true,
        height: true,
      })
    </script>

    <template>
      <div :class="classObject"></div>
    </template>

    <style>
    .active {
      background-color: red;
    }

    .width {
      width: 500px;

    }

    .height {
      height: 60px;
    }
    </style>

    绑定返回一个对象的计算属性

    <script setup>
    import { computed, ref } from 'vue'

    const isActive = ref(true)
    const isWidth = ref(true)
    const isHeight = ref(true)

    const classObject = computed(() => {
      return {
        active: isActive.value,
        width: isWidth,
        height: isHeight,
      }
    })

    </script>

    <template>
      
      <div :class="classObject"></div>

    </template>

    <style>
    .active {
      background-color: red;
    }

    .width {
      width: 500px;

    }

    .height {
      height: 60px;
    }
    </style>

    绑定数组

    <script setup>
    import { computed, ref } from 'vue'

    const activeClass = ref('active')
    const sizeClass = ref('size')

    </script>

    <template>
      <!-- 三元表达式 -->
      <div :class="[true ? activeClass : '', sizeClass]"></div>
    </template>

    <style>
    .active {
      background-color: red;
    }

    .size {
      width: 500px;
      height: 60px;
    }
    </style>

    父组件传递给子组件样式

    只有一个根元素

    父组件

    <script setup>
    import { ref } from 'vue'
    import ChildComp from './components/ChildComp.vue'
    const isSize = ref(true)

    </script>

    <template>
      <ChildComp class="parent" :class="{ size: isSize }" />
    </template>

    <style>
    .parent {
      background-color: aqua;
    }

    .size {
      width: 500px;
      height: 60px;
    }
    </style>

    子组件

    <template>
      <div></div>
    </template>

    </template>
    绑定内联样式
    绑定对象

    绑定对象值

    <script setup>
    import { ref } from 'vue'

    const font_color = ref('aqua')
    const font_size = ref(30)
    </script>

    <template>
      <div :style="{ 'font-size': font_size + 'px', 'color': font_color }">it's mygo!!!!!</div>
    </template>

    绑定对象

    <script setup>
    import { ref } from 'vue'

    const styleObject = ref({
      'font-size': '30px',
      'color': 'aqua',
    })
    </script>

    <template>
      <div :style="styleObject">it's mygo!!!!!</div>
    </template>

    绑定对象数组

    <script setup>
    import { ref } from 'vue'

    const fontColor = ref({ color: 'aqua' })
    const fontSize = ref({ fontSize: '30px' })
    </script>

    <template>
      <!-- 样式 对象 数组 -->
      <div :style="[fontColor, fontSize]">it's mygo!!!!!</div>
    </template>
    样式多值

    可以对一个样式属性提供多个 (不同前缀的) 值，举例来说：

    <div :style="{ display: ['-webkit-box', '-ms-flexbox', 'flex'] }"></div>
    数组仅会渲染浏览器支持的最后一个值。在这个示例中， 在支持不需要特别前缀的浏览器中都会渲染为 `display: flex`。
