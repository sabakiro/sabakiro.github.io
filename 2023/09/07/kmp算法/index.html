kmp算法

#include<stdio.h>
#include<stdlib.h>
#include<string.h>
void getNext(int next[], char* des) {
	int k = -1;
	int j = 0;
	next[0] = -1;
	int len = strlen(des);
	//abcabd
	//next[-1,0,0,1,2,0]
	while (j < len)
	{
		if (k == -1 || des[k] == des[j]) {
			k++;
			j++;
			next[j] = k;
		}
		else {
			k = next[k];
		}
	}
}
//核心思想：目标字符串不回退，字串回退
int kmp(char* str1, char* str2) {
	int len1 = strlen(str1);
	int len2 = strlen(str2);
	int i = 0;
	int j = 0;
	int* next = malloc(sizeof(int) * len2);
	//记录str2的字符分布规律
	getNext(next, str2);
	while (i < len1) {
		if (j == -1 || str1[i] == str2[j]) {
			i++;
			j++;
			if (j >= len2) {
				return i - j;
			}
		}
		else {
			j = next[j];
		}
	}
	return -1;
}
