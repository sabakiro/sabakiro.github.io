使用引用来模仿指针

static B: [u8; 10] = [99, 97, 114, 114, 121, 116, 111, 119, 101, 108];
static C: [u8; 11] = [116, 104, 97, 110, 107, 115, 102, 105, 115, 104, 0];
fn main() {
    let a = 42;
    let b = &B;
    let c = &C;
    //{:p}这个语法告诉Rust，把变量当作指针来执行格式化，
    //也就说，要求输出该值所指向的内存地址
    println!("a:{},b:{:p},c:{:p}", a, b, c);
    //a:42,b:0x7ff69eefd370,c:0x7ff69eefd37a
}

通过指向几种不同的类型，来对比引用和Box<T>

use std::mem::size_of;

static B: [u8; 10] = [99, 97, 114, 114, 121, 116, 111, 119, 101, 108];
static C: [u8; 11] = [116, 104, 97, 110, 107, 115, 102, 105, 115, 104, 0];
fn main() {
    /*
    usize的实际大小就是内存地址的宽度，
    而内存地址的宽度取决于编译代码时使用的CPU
    所以此CPU又叫作编译目标
     */
    let a = 42;
    /*
    &[u8;10]读作“一个10字节的数组的引用”。
    此数组存放在内存的静态区域中
    而该引用本身（一个宽度为uszie字节的指针）是存放在栈上的
     */
    let b = &B;
    /*
    Box<[u8]>类型是一个装箱的字节切片。
    当程序员把某个值装入一个箱子(box)中后，
    该值的所有权就被转移给了箱子的所有者
     */
    let c = Box::new(C);

    println!("a (an unsigned interger):");
    println!(" location: {:p}", &a);
    println!(" size:     {:?} bytes", size_of::<usize>());
    println!(" value:    {:?}", a);
    println!();
    println!("b (a reference to B):");
    println!("  location:   {:p}", &b);
    println!("  size:       {:?} bytes", size_of::<&[u8; 10]>());
    println!("  points to:  {:p}", b);
    println!();
    println!("c (a 'box' for C):");
    println!("  location:   {:p}", &c);
    println!("  size:       {:?} bytes", size_of::<&[Box<[u8]>]>());
    println!("  points to:  {:p}", c);
    println!();
    println!("B (an array of 10 bytes):");
    println!("  location:   {:p}", &B);
    println!("  size:       {:?} bytes", size_of::<&[u8; 10]>());
    println!("  points to:  {:?}", B);
    println!();
    println!("C (an array of 11 bytes):");
    println!("  location:   {:p}", &C);
    println!("  size:       {:?} bytes", size_of::<&[u8; 11]>());
    println!("  points to:  {:?}", C);
    println!();
}
/*
a (an unsigned interger):
 location: 0x85f06fed6c
 size:     8 bytes
 value:    42

b (a reference to B):
  location:   0x85f06fed70
  size:       8 bytes
  points to:  0x7ff6edc1f390

c (a 'box' for C):
  location:   0x85f06fed78
  size:       16 bytes
  points to:  0x285e35797c0

B (an array of 10 bytes):
  location:   0x7ff6edc1f390
  size:       8 bytes
  points to:  [99, 97, 114, 114, 121, 116, 111, 119, 101, 108]

C (an array of 11 bytes)::
  location:   0x7ff6edc1f39a
  size:       8 bytes
  points to:  [116, 104, 97, 110, 107, 115, 102, 105, 115, 104, 0]
*/

输出来自外部来源的字符串

/*
一种智能指针类型，
能够从其指针位置读取数据而无须先复制它
*/
use std::borrow::Cow;
/*
CStr是一个类似于C字符串的类型
它让Rust能够读取以0作为结束标志的字符串
*/
use std::ffi::CStr;
/*
c_char是Rust中i8类型的别名，
但对于特定的平台，
它可能会存在一些细微的差别
*/
use std::os::raw::c_char;

static B: [u8; 10] = [99, 97, 114, 114, 121, 116, 111, 119, 101, 108];
static C: [u8; 11] = [116, 104, 97, 110, 107, 115, 102, 105, 115, 104, 0];
fn main() {
    /*
    在这里引入每个变量，在后面的println！中才能够访问到。
    假如把创建b和c的两行代码放到下面的unsafe块中，
    那么在后面超出作用域后，
    输出语句那里就访问不到了
     */
    let a = 42;
    /*
    String是一种智能指针类型，
    它包含一种指向后端数组的指针和一个用于存储其大小的字段
     */
    let b: String;
    /*
    Cow就收一个类型参数，即它所指向的数据类型。
    str是CStr.to_string_lossy() 的返回类型，
    所以放在这里是合适的
     */
    let c: Cow<str>;
    unsafe {
        /*
        引用不能直接转换为*mut T，
        而后者正是Stirng::from_raw_parts()所需要的类型。
        但是，*const T可以转换为*mut T，
        所以就有了这个二次类型转换的语法
         */
        let b_ptr = &B as *const u8 as *mut u8;
        /*
        String::from_raw_parts()接受3个参数，
        一个参数是指向字节数组的指针(*mut T),
        另两个是大小和容量参数
         */
        b = String::from_raw_parts(b_ptr, 10, 10);
        /*
        我们把一个*const u8转换为*const i8，
        后者就是c_char的别名。在这里，
        转换到i8之所以能够成功，是因为依据ASCII标准，
        数据一定都是小于128的
         */
        let c_ptr = &C as *const u8 as *const c_char;
        /*
        从概念上讲，Cstr::from（）负责读取指针指向的数据，
        直到遇到0为止，
        然后利用读取到的结果数据生成一个Cow<str>
         */
        c = CStr::from_ptr(c_ptr).to_string_lossy();
    }
    println!("a:{}, b:{}, c:{} ", a, b, c);
}
//a:42, b:carrytowel, c:thanksfish

rust中的原始指针

创建一个原始指针(*const T)

fn main() {
    let a:i64=42;
    //把a的引用（&a）转换为指向i64的常量原始指针（*const i64）
    let a_ptr=&a as *const i64;
    //输出a的值（42）以及它在内存中的地址
    println!("a :{} ({:p})",a,a_ptr);
}
//a :42 (0x35f918f5e8)

识别一个值的地址

fn main() {
    /*
    展示了如何将一个引用转换为原始指针，
    然后使用std ::mem :: transmute()来识别这个值的地址
     */
    let a: i64 = 42;
    let a_ptr = &a as *const i64;
    let a_addr: usize = unsafe { std::mem::transmute(a_ptr) };
    println!("a:{} ({:p}.....0x{:x})", a, a_ptr, a_addr + 7);
}
//a:42 (0xee38b9f8b8.....0xee38b9f8bf)

使用windows api来查看内存
依赖项

[dependencies]
#定义了一些有用的类型别名。
winapi="0.2"
#提供了使用KERNEL.DLL来与Windows API进行交互的功能。
kernel32-sys="0.2"
use kernel32;
use winapi;
use winapi::{
    //在Rust里，这个类型叫作u32。
    DWORD,
    /* ::os::raw::c_void定义了void指针。
    一个HANDLE（句柄）是在Windows中指向一些不透明资源的指针。
    */
    HANDLE,
    // 指向一个SYSTEM_INFO结构体的指针。
    LPSYSTEM_INFO,
    LPVOID,
    MEMORY_BASIC_INFORMATION as MEMINFO,
    /*
    在Windows里，数据类型的名称常常会用一个前缀代表类型的缩写。
    P代表指针，LP代表长指针（例如64位的指针）
     */
    PVOID,
    SIZE_T,
    //在Windows内部定义的一些结构体。
    SYSTEM_INFO,
};
/*这些变量的初始化代码会放在unsafe块中。
为了使其在外部作用域中能够被访问到，所以需要在这里定义。
*/
fn main() {
    let this_pid: DWORD;
    let this_proc: HANDLE;
    let min_addr: LPVOID;
    let max_addr: LPVOID;
    let mut base_addr: PVOID;
    let mut proc_info: SYSTEM_INFO;
    let mut mem_info: MEMINFO;
    const MEMINFO_SIZE: usize = std::mem::size_of::<MEMINFO>();
    unsafe {
        //这个代码块保证了所有内存被初始化。

        base_addr = std::mem::zeroed();
        proc_info = std::mem::zeroed();
        mem_info = std::mem::zeroed();
    }
    unsafe {
        //在这个代码块中，是发起系统调用的代码。

        this_pid = kernel32::GetCurrentProcessId();
        this_proc = kernel32::GetCurrentProcess();
        kernel32::GetSystemInfo(
            /*此函数没有使用返回值，而是使用了C的惯用法，
            来把运行结果提供给调用方。我们提供一个指向某个预定义结构体的指针，
            然后在函数返回后读取该结构体的新值，以查看其结果。
             */
            &mut proc_info as LPSYSTEM_INFO,
        );
    };
    /*
    为了方便，此处重命名了这些变量。
     */
    min_addr = proc_info.lpMinimumApplicationAddress;
    max_addr = proc_info.lpMaximumApplicationAddress;
    println!("{:?} @ {:p}", this_pid, this_proc);
    println!("{:?}", proc_info);
    println!("min: {:p}, max: {:p}", min_addr, max_addr);
    loop {
        //这个循环完成了扫描地址空间的工作。

        let rc: SIZE_T = unsafe {
            kernel32::VirtualQueryEx(
                /*这个系统调用，
                提供了这个正在运行的程序的内存地址空间中一个指定段的信息，
                此指定段的起始位置的地址是base_addr。
                 */
                this_proc,
                base_addr,
                &mut mem_info,
                MEMINFO_SIZE as SIZE_T,
            )
        };
        if rc == 0 {
            break;
        }
        println!("{:#?}", mem_info);
        base_addr = ((base_addr as u64) + mem_info.RegionSize) as PVOID;
    }
}
/*
操作系统提供了用于读取和写入内存的工具，甚至在其他程序中也可以完成:
伪代码：
let pid = some_process_id;
OpenProcess(pid);
loop address space {
    *call* VirtualQueryEx() to access the next memory segment
    *scan* the segment by calling ReadProcessMemory(),
    looking for a selected pattern
    *call* WriteProcessMemory() with the desired value
}
*/
