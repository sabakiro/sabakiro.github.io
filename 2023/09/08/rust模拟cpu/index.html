从操作码中提出多个变量

fn function() {
    let opcode: u16 = 0x71E4;
    /*
    先使用与（&）操作进行过滤，把想要的那个半字节保留下来，
    然后把这个半字节的位数据右移位到最低有效位的位置上。
    使用十六进制表示只是出于方便，每个十六进制的数字位表示二进制的4个位。
    一个OxF的值就选中了单个半字节所有的位数据
     */
    let c = (opcode & 0xF000) >> 12;
    let x = (opcode & 0x0F00) >> 8;
    let y = (opcode & 0x00F0) >> 8;
    let d = (opcode & 0x000F) >> 8;
    /*
    opcode中的这4个半字节，在处理以后会拆分出4个独立的变量
    */
    assert_eq!(c, 0x7);
    assert_eq!(x, 0x1);
    assert_eq!(y, 0xE);
    assert_eq!(d, 0x4);
    /*
    如果想要多个半字节，
    可以通过增加过滤器的宽度来一次性选中更多的半字节。
    为此，在这里向右移位就不是必需的了。
     */
    let nnn = opcode & 0x0FFF;
    let kk = opcode & 0x00FF;
    assert_eq!(nnn, 0x1E4);
    assert_eq!(kk, 0xE4);
}

加法器的原型模拟器

struct CPU {
    current_operation: u16,
    registers: [u8; 2],
}
impl CPU {
    fn read_opcode(&self) -> u16 {
        self.current_operation
    }
    fn run(&mut self) {
        //loop{ 先不循环
        let opcode = self.read_opcode();
        let c = ((opcode & 0xF000) >> 12) as u8;
        let x = ((opcode & 0x0F00) >> 8) as u8;
        let y = ((opcode & 0x00F0) >> 4) as u8;
        let d = ((opcode & 0x000F) >> 0) as u8;
        match (c, x, y, d) {
            //把真正执行此操作的任务分配到负责执行该操作的“硬件电路”上
            (0x8, _, _, 0x4) => self.add_xy(x, y),
            
            _ => todo!("opcode {:04x}", opcode),
        }
        //} 
    }
    fn add_xy(&mut self, x: u8, y: u8) {
        self.registers[x as usize] += self.registers[y as usize];
    }
}
fn main() {
    //-------初始化cpu---------
    let mut cpu = CPU {
        current_operation: 0,
        registers: [0; 2],
    };
    /*
    常量0x8014是CPU将要解码的操作码:
    ● 8表示此操作需要使用两个寄存器。
    ● 0对应cpu.registers[0]，即寄存器0。
    ● 1对应cpu.registers[1]，即寄存器1。
    ● 4表示此操作是加法运算。
     */
    cpu.current_operation = 0x8014;
    cpu.registers[0] = 5;
    cpu.registers[1] = 10;
    //-------初始化cpu---------
    cpu.run();
    assert_eq!(cpu.registers[0], 15);
    println!("5 + 10 = {}", cpu.registers[0]);
}

累加器

struct CPU {
    registers: [u8; 16],
    /*
    使用usize而不是16与最初的初范是不同的
    但是我们使用的usize，因为Rust允许此类型被用于索引
     */
    position_in_memory: usize,
    memory: [u8; 0x1000],
}
impl CPU {
    /*
    内存中读出一个操作码，此过程需要把两个u8的值合并为一个u16。
     */
    fn read_opcode(&self) -> u16 {
        let p = self.position_in_memory;

        let op_byte1 = self.memory[p] as u16;
        let op_byte2 = self.memory[p + 1] as u16;

        //要创建一个u16类型的操作码,我们使用 逻辑或 操作,
        //把内存中的两个值合并到一起.这两个值需要先转换为u16,
        //如果不先做这个转换,左移位会将所有的位数据都设为0
        op_byte1 << 8 | op_byte2
    }
    fn run(&mut self) {
        loop {
            //持续执行,而不是只执行一条指令
            let opcode = self.read_opcode();
            //自增position_in_memory,指向下一条指令
            self.position_in_memory += 2;

            let c = ((opcode & 0xF000) >> 12) as u8;
            let x = ((opcode & 0x0F00) >> 8) as u8;
            let y = ((opcode & 0x00F0) >> 4) as u8;
            let d = ((opcode & 0x000F) >> 0) as u8;
            match (c, x, y, d) {
                // 当遇到的操作码为0x0000时,
                //此处的短路功能会终止函数的执行
                (0, 0, 0, 0) => {
                    return;
                }
                //把真正执行此操作的任务分配到负责执行该操作的“硬件电路”上
                (0x8, _, _, 0x4) => self.add_xy(x, y),

                _ => todo!("opcode {:04x}", opcode),
            }
        }
    }
    fn add_xy(&mut self, x: u8, y: u8) {
        let arg1 = self.registers[x as usize];
        let arg2 = self.registers[y as usize];
        /*
        在CHIP-8中，最后一个寄存器被用作进位标志。
        如果进位标志被设置，这表明有一个操作在u8大小的寄存器中的产生了溢出

        对于u8 类型来说,overflowing_add()方法的返回值类型为(u8,bool)
        如果检测到溢出则返回值这个布尔类型的值为true
         */
        let (value, overflow) = arg1.overflowing_add(arg2);
        self.registers[x as usize] = value;
        if overflow {
            self.registers[0xF] = 1;
        } else {
            self.registers[0xF] = 0;
        }
    }
}
fn main() {
    let mut cpu = CPU {
        registers: [0; 16],
        memory: [0; 4096],
        position_in_memory: 0,
    };
    //使用值来初始化寄存器
    cpu.registers[0] = 5;
    cpu.registers[1] = 10;
    cpu.registers[2] = 10;
    cpu.registers[3] = 10;

    let mem = &mut cpu.memory;
    //加载操作码0x8014,0x8014意思是把寄存器1的值加到寄存器0上
    mem[0] = 0x80;
    mem[1] = 0x14;
    //加载操作码0x8024,0x8024意思是把寄存器2的值加到寄存器0上
    mem[2] = 0x80;
    mem[3] = 0x24;
    //加载操作码0x8034,0x8034意思是把寄存器3的值加到寄存器0上
    mem[4] = 0x80;
    mem[5] = 0x34;

    cpu.run();
    assert_eq!(cpu.registers[0], 35);
    println!("5 + 10 + 10 + 10 = {}", cpu.registers[0]);
}
