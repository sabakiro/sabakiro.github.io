使用一个自旋循环策略来暂停20ms

use std::{thread, time};
fn main() {
    for n in 1..1001 {
        let mut handlers: Vec<thread::JoinHandle<()>> = Vec::with_capacity(n);
        let start = time::Instant::now();
        let pause = time::Duration::from_millis(20);
        for _m in 0..n {
            let handle = thread::spawn(move || {
                let start = time::Instant::now();
                while start.elapsed() < pause {
                    //使用thread::yield_now()来控制出让
                    thread::yield_now();
                }
            });
            handlers.push(handle);
        }

        while let Some(handle) = handlers.pop() {
            handle.join();
        }

        let finish = time::Instant::now();
        println!("{}\t{:02?}", n, finish.duration_since(start));
    }
}

svg="0.6"
use std::env;
use svg::node::element::path::{Command, Data, Position};
use svg::node::element::{Path, Rectangle};
use svg::Document;

/*枚举体类型Operation和Orientation是在后面的代码中定义的。
在这里使用use关键字把它们都导入进来，可以去除源代码中的大量干扰。
*/
use crate::Operation::{Forward, Home, Noop, TurnLeft, TurnRight};
use crate::Orientation::{East, North, South, West};
//HEIGHT和WIDTH提供了绘图的边界。
const WIDTH: isize = 400;
const HEIGHT: isize = WIDTH;

/*常量HOME_Y和HOME_X让我们可以轻松地重置绘制的位置。
这里的Y是纵坐标，X是横坐标。
*/
const HOME_Y: isize = HEIGHT / 2;
const HOME_X: isize = WIDTH / 2;

//STROKE_WIDTH用于输出SVG的一个参数，定义绘制出的线条的粗细。
const STROKE_WIDTH: usize = 5;

#[derive(Debug, Clone, Copy)]
enum Orientation {
    //在这里使用的是描述而不是数值，这样就避免了数学运算。
    North,
    East,
    West,
    South,
}

#[derive(Debug, Clone, Copy)]
// 要想生成更丰富的输出，可以为你的程序再扩展一些可用的操作。
enum Operation {
    /*使用isize，让我们可以扩展这个示例代码，
    实现一个反方向的移动操作，而且不需要为此枚举体添加新的变体。
     */
    Forward(isize),
    TurnLeft,
    TurnRight,
    Home,
    /*如果我们遇到非法输入，就使用Noop。为了写入错误消息，
    这里我们保留了那个非法的字节。
    */
    Noop(u8),
}

#[derive(Debug)]
// Artist结构体维护着当前的状态。
struct Artist {
    x: isize,
    y: isize,
    heading: Orientation,
}

impl Artist {
    fn new() -> Artist {
        Artist {
            heading: North,
            x: HOME_X,
            y: HOME_Y,
        }
    }

    fn home(&mut self) {
        self.x = HOME_X;
        self.y = HOME_Y;
    }
    /*forward()在match表达式的里面修改了self。
    与之形成对照的是，
    turn_left() 和turn_right()在match表达式的外面修改了self。
     */
    fn forward(&mut self, distance: isize) {
        match self.heading {
            North => self.y += distance,
            South => self.y -= distance,
            West => self.x += distance,
            East => self.x -= distance,
        }
    }

    fn turn_right(&mut self) {
        self.heading = match self.heading {
            North => East,
            South => West,
            West => North,
            East => South,
        }
    }
    /*forward()在match表达式的里面修改了self。与之形成对照的是，
    turn_left() 和turn_right()在match表达式的外面修改了self。
    */
    fn turn_left(&mut self) {
        self.heading = match self.heading {
            North => West,
            South => East,
            West => South,
            East => North,
        }
    }

    //wrap() 确保了图形的绘制是保持在边界之内的。
    fn wrap(&mut self) {
        if self.x < 0 {
            self.x = HOME_X;
            self.heading = West;
        } else if self.x > WIDTH {
            self.x = HOME_X;
            self.heading = East;
        }

        if self.y < 0 {
            self.y = HOME_Y;
            self.heading = North;
        } else if self.y > HEIGHT {
            self.y = HOME_Y;
            self.heading = South;
        }
    }
}

fn parse(input: &str) -> Vec<Operation> {
    let mut steps = Vec::<Operation>::new();
    for byte in input.bytes() {
        let step = match byte {
            b'0' => Home,
            b'1'..=b'9' => {
                /*
                在ASCII字符的编码中，数字的部分是从0x30（十进制的48）开始的。
                所以此代码(byte–0x30)会把b'2' 转换成2。
                要在整个u8的范围内来执行这样的操作有可能会引发恐慌，
                但是这行代码是安全的，这要归功于模式匹配所提供的保证。
                 */
                let distance = (byte - 0x30) as isize;
                Forward(distance * (HEIGHT / 10))
            }
            /*
            这里还留有很多的机会，可以添加更多的指令，
            能够生成更多为精细的图形，
            同时还不会解析过程的复杂性
            */
            b'a' | b'b' | b'c' => TurnLeft,
            b'd' | b'e' | b'f' => TurnRight,
            
            /*虽然我们不期望任何非法的字符，但在输入流中可能会有非法字符。
            在这里使用了Noop操作，让我们可以把解析的过程和产生输出的过程分离开来。
            */
            _ => Noop(byte),
        };
        steps.push(step);
    }
    steps
}
/*
把从parse()中得到的Vec<Operation>转换成Vec<Command>。
该函数的输出结果稍后会被用来生成一张SVG的图。
*/
fn convert(operations: &Vec<Operation>) -> Vec<Command> {
    let mut turtle = Artist::new();

    let mut path_data = Vec::<Command>::with_capacity(operations.len());
    /*
    开始，先把海龟置于绘图区的中心位置
    */
    let start_at_home = Command::Move(Position::Absolute, (HOME_X, HOME_Y).into());
    path_data.push(start_at_home);

    for op in operations {
        match *op {
	        //不是马上就生成相应的命令，而是先修改海龟的内部状态
            Forward(distance) => turtle.forward(distance),
            TurnLeft => turtle.turn_left(),
            TurnRight => turtle.turn_right(),
            Home => turtle.home(),
            Noop(byte) => {
                eprintln!("warning: illegal byte encountered: {:?}", byte);
            }
        };
		/*创建一个Command::Line
		（与海龟当前的坐标位置相连接的一条直线）
        */
        let path_segment = Command::Line(Position::Absolute, (turtle.x, turtle.y).into());
        path_data.push(path_segment);

		//如果海龟出界了，就把它送回到中心位置
        turtle.wrap();
    }
    path_data
}

fn generate_svg(path_data: Vec<Command>) -> Document {
    let background = Rectangle::new()
        .set("x", 0)
        .set("y", 0)
        .set("width", WIDTH)
        .set("height", HEIGHT)
        .set("fill", "#ffffff");

    let border = background
        .clone()
        .set("fill-opacity", "0.0")
        .set("stroke", "#cccccc")
        .set("stroke-width", 3 * STROKE_WIDTH);

    let sketch = Path::new()
        .set("fill", "none")
        .set("stroke", "#2f2f2f")
        .set("stroke-width", STROKE_WIDTH)
        .set("stroke-opacity", "0.9")
        .set("d", Data::from(path_data));

    let document = Document::new()
        .set("viewBox", (0, 0, HEIGHT, WIDTH))
        .set("height", HEIGHT)
        .set("width", WIDTH)
        .set("style", "style=\"outline: 5px solid #800000;\"")
        .add(background)
        .add(sketch)
        .add(border);

    document
}

fn main() {
    let args = env::args().collect::<Vec<String>>();
    let input = args.get(1).unwrap();
    let default_filename = format!("{}.svg", input);
    let save_to = args.get(2).unwrap_or(&default_filename);

    let operations = parse(input);
    let path_data = convert(&operations);
    let document = generate_svg(path_data);
    svg::save(save_to, &document).unwrap();
}

新的render-hex：此版本使用了基于通道来实现的任务队列

svg="0.6"
crossbeam="0.7"
use std::thread;
use crossbeam::channel::unbounded;
enum Work {
    //创建一个消息的类型，我们会使用通道来发送这种类型的消息。

	/*
	 //这个元组中的usize字段标明了已处理完的字节的顺序位置。
	 需要这样一个字段，是因为返回的任务结果很有可能是无序的。
	*/
    Task((usize, u8)),
    
     //给worker线程一个标志性的消息，指示该线程是时候该关闭了。
    Finished,         
}

fn parse_byte(byte: u8) -> Operation {
    //把worker需要执行的功能抽出来，以便简化程序逻辑。
    match byte {
        b'0' => Home,
        b'1'..=b'9' => {
            let distance = (byte - 0x30) as isize;
            Forward(distance * (HEIGHT / 10))
        }
        b'a' | b'b' | b'c' => TurnLeft,
        b'd' | b'e' | b'f' => TurnRight,
        _ => Noop(byte),
    }
}

fn parse(input: &str) -> Vec<Operation> {
    let n_threads = 2;
    // 给待完成的任务创建一个通道。
    let (todo_tx, todo_rx) = unbounded();
     
     // 为将要返回的解码后的指令创建一个通道。
    let (results_tx, results_rx) = unbounded();
    let mut n_bytes = 0;
    for (i, byte) in input.bytes().enumerate() {
    //把待完成的工作填充到此任务队列中。
        todo_tx.send(Work::Task((i, byte))).unwrap(); 
        //这句代码用来跟踪待完成的任务数量。
        n_bytes += 1; 
    }

    for _ in 0..n_threads {
        //给所有worker线程都发送一个信号，通知这些线程是时候关闭了。
        todo_tx.send(Work::Finished).unwrap();
    }

    for _ in 0..n_threads {
    //通过使用克隆，让通道可以跨线程共享。
        let todo = todo_rx.clone(); 
        let results = results_tx.clone();
        thread::spawn(move || loop {
            let task = todo.recv();
            let result = match task {
                Err(_) => break,
                Ok(Work::Finished) => break,
                Ok(Work::Task((i, byte))) => (i, parse_byte(byte)),
            };
            results.send(result).unwrap();
        });
    }
    /*
    由于返回的任务结果可以是任意的顺序，
    因此我们先初始化一个对应完整任务数量的Vec<command>，
    稍后将会用传进来的任务结果覆盖它。
    我们使用了Vec而没有使用数组，
    是因为此函数的类型签名已经是这样的了，
    我们并不想再重构整个程序来适应新的修改。
    */
    let mut ops = vec![Noop(0); n_bytes];
    for _ in 0..n_bytes {
        let (i, op) = results_rx.recv().unwrap();
        ops[i] = op;
    }
    ops
}
