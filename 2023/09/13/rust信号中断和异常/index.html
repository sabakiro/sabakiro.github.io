rust信号中断和异常

一个接收SIGSTOP和SIGCONT的基本应用程序

use core::time;
use std::process;

fn main() {
    let delay = time::Duration::from_secs(1);
    let pid = process::id();
    println!("{}", pid);
    for i in 1..=60 {
        std::thread::sleep(delay);
        println!(".{}", i);
    }
}
#中止
kill -SIGSTOP pid
#恢复
kill -SIGCONT pid

创建一个信号处理程序，用于修改一个全局变量

libc="0.2"
#![cfg(not(windows))] //出了当前代码不能运行在Windows上。
use libc::{SIGTERM, SIGUSR1};
use std::thread::sleep;
use std::time::Duration;
static mut SHUT_DOWN: bool = false;

fn register_signal_handlers() {
    unsafe {
        libc::signal(SIGTERM, handle_sigterm as usize);
        libc::signal(SIGUSR1, handle_sigusr1 as usize);
    }
}

#[allow(dead_code)] //如果不加这个属性，rustc会发出警告，通知这些函数从未被调用过。
fn handle_sigterm(_signal: i32) {
    register_signal_handlers(); //尽快地重新注册信号，能够尽可能地防止由于信号的变化而影响信号处理程序本身。

    println!("SIGTERM");

    unsafe {
        //修改可变静态变量是不安全的。
        SHUT_DOWN = true;
    }
}

#[allow(dead_code)]
fn handle_sigusr1(_signal: i32) {
    register_signal_handlers();

    println!("SIGUSR1");
}

//初始化程序并通过主循环进行迭代。
fn main() {
/*
register_signal_handlers()函数：通过libc与操作系统进行沟通，
每个信号的信号处理程序是什么。此函数使用了一个函数指针，
函数指针把函数看作数据

必须尽早执行，否则信号将不能得到正确的处理。*/
    register_signal_handlers(); 

    let delay = Duration::from_secs(1);

    for i in 1_usize.. {
        println!("{}", i);
        unsafe {
            //对可变静态变量的访问是不安全的。
            if SHUT_DOWN {
                println!("*");
                return;
            }
        }

        sleep(delay);

        let signal = if i > 2 { SIGTERM } else { SIGUSR1 };

        unsafe {
            //对libc函数的调用是不安全的，这些函数所产生的影响是Rust无法控制的。
            libc::raise(signal);
        }
    }
    unreachable!();
}
/*
1
SIGUSR1
2
SIGUSR1
3
SIGTERM
4
*
*/

使用libc::SIG_IGN来忽略信号

libc="0.2"
use libc::{raise, signal};
use libc::{SIGTERM, SIG_DFL, SIG_IGN};
fn main() {
// 这里需要使用unsafe块，因为Rust控制不了在函数边界以外所发生的事情。
    unsafe {
        // 忽略SIGTERM信号。
        signal(SIGTERM, SIG_IGN); 
        //libc::raise() 让代码可以发出信号。在本例中，这个信号是发给自己的。
        raise(SIGTERM); 
    }
    println!("ok");

    unsafe {
	    //把SIGTERM信号的行为重置成默认值。
        signal(SIGTERM, SIG_DFL); 
        
        //这会让此程序终止。
        raise(SIGTERM); 
    }
	//这行代码永远不会运行，所以这个字符串是不会输出的。
    println!("not ok"); 
}
/*
ok
Terminated
*/

sjlj项目的元数据

[dependencies]
libc="0.2"
#![feature(link_llvm_intrinsics)]
#![allow(non_camel_case_types)]
//只在支持的平台上进行编译。
#![cfg(not(windows))]
use libc::{SIGALRM, SIGHUP, SIGQUIT, SIGTERM, SIGUSR1};
use std::mem;
const JMP_BUF_WIDTH: usize = mem::size_of::<usize>() * 8;
type jmp_buf = [i8; JMP_BUF_WIDTH];

//如果是true，则程序退出。
static mut SHUT_DOWN: bool = false;
static mut RETURN_HERE: jmp_buf = [0; JMP_BUF_WIDTH];

//允许递归的深度是3。
const MOCK_SIGNAL_AT: usize = 3;

extern "C" {
    #[link_name = "llvm.eh.sjlj.setjmp"]
    pub fn setjmp(_: *mut i8) -> i32;
	
	/*
	LLVM编译器访问操作系统的longjmp()功能。
	longjmp()允许程序跳出所在的栈帧并跳转到它的地址空间内的任何的位置。
	*/
    #[link_name = "llvm.eh.sjlj.longjmp"]
    pub fn longjmp(_: *mut i8);
}

//#[inline]属性会把函数标记为可用于内联，这是一种可以消除函数调用开销的编译器优化技术。
#[inline]
fn ptr_to_jmp_buf() -> *mut i8 {
    unsafe { &RETURN_HERE as *const i8 as *mut i8 }
}

#[inline]
fn return_early() {
    let franken_pointer = ptr_to_jmp_buf();

    // 这是不安全的，是因为Rust不能保证LLVM在RETURN_HERE处对内存所做的事情。
    unsafe { longjmp(franken_pointer) };
}

fn register_signal_handler() {
    unsafe {
        //要求libc::signal()将handle_signals()函数与SIGUSR1信号进行关联。
        libc::signal(SIGUSR1, handle_signals as usize);
    }
}

#[allow(dead_code)]
fn handle_signals(sig: i32) {
    register_signal_handler();

    let should_shut_down = match sig {
        SIGHUP => false,
        SIGALRM => false,
        SIGTERM => true,
        SIGQUIT => true,
        SIGUSR1 => true,
        _ => false,
    };

    unsafe {
        SHUT_DOWN = should_shut_down;
    }

    return_early();
}

fn print_depth(depth: usize) {
    for _ in 0..depth {
        print!("#");
    }
    println!();
}

fn dive(depth: usize, max_depth: usize) {
    unsafe {
        if SHUT_DOWN {
            println!("!");
            return;
        }
    }
    print_depth(depth);

    if depth >= max_depth {
        return;
    } else if depth == MOCK_SIGNAL_AT {
        unsafe {
            libc::raise(SIGUSR1);
        }
    } else {
        dive(depth + 1, max_depth);
    }
    print_depth(depth);
}

fn main() {
    const JUMP_SET: i32 = 0;

    register_signal_handler();

    let return_point = ptr_to_jmp_buf();
    let rc = unsafe { setjmp(return_point) };
    if rc == JUMP_SET {
        dive(0, 10);
    } else {
        println!("early return!");
    }

    println!("finishing!")
}
