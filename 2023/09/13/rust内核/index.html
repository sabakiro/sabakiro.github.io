安装所需要的工具

#cargo-binutils包，允许cargo通过子命令来直接操作可执行文件，
#而这些子命令是由Rust构建并由cargo安装的一些实用程序。
#使用cargo-binutils而不是通过其他方式来安装binutils，
#是为了防止潜在的版本不匹配问题出现。
cargo install cargo-binutils

#允许cargo构建一个引导映像(boot image)。
#引导映像是一个可以直接在硬件上启动的可执行文件。
cargo install bootimage

#夜间版工具链(nightly toolchain)，安装Rust编译器的夜间版，
# 可以解锁尚未被标记为稳定版的那些功能，因此这也限制了Rust向后兼容的保证。
# 在本章中，我们将会访问到编译器内部的一些东西。
# 目前这些东西还不太可能稳定下来。在本章中，我们把夜间版设置成默认工具链，
# 这样可以简化项目的构建步骤。要还原这项设置，
# 可以使用命令rustup default stable。
rustup toolchain install nightly

rustup default nighty

# rust-src组件，下载Rust编程语言的源代码，
#这允许Rust为新的操作系统编译一个编译器。
rustup component add rust-src

#llvm-tools-preview组件，安装LLVM编译器的扩展，它是Rust编译器的一部分。
rustup component add llvm-tools-preview

#[no_mangle]
pub extern "C" fn _start() -> ! {
    ////////VGA帧缓冲区的修改////////////
    let framebuffer = 0xb8000 as *mut u8;

    unsafe {
        //把内存位置0x8001中的数据设为0x30
        framebuffer
            //加上1以后，此指针的地址是0xb8001。
            .offset(1)
            //把背景设置成青色。
            .write_volatile(0x30);
    }
    ////////VGA帧缓冲区的修改////////////

    loop {
        hlt() //可以省电
    }
}

fledgeos-3:现在Fledgeos可以在屏幕上输出文本

#![feature(core_intrinsics)]
#![feature(lang_items)]
#![no_std]
#![no_main]
use core::intrinsics;
use core::panic::PanicInfo;
use x86_64::instructions::hlt;

#[allow(unused)]
#[derive(Clone, Copy)]
#[repr(u8)]
enum Color {
    Black = 0x0,
    White = 0xF,
    Blue = 0x1,
    BrightBlue = 0x9,
    Green = 0x2,
    BrightGreen = 0xA,
    Cyan = 0x3,
    BrightCyan = 0xB,
    Red = 0x4,
    BrightRed = 0xC,
    Magenta = 0x5,
    BrightMagenta = 0xD,
    Brown = 0x6,
    Yellow = 0xE,
    Gray = 0x7,
    DarkGray = 0x8,
}

////////Cursor的定义和所实现的方法//////////
struct Cursor {
    position: isize,
    foreground: Color,
    background: Color,
}

impl Cursor {
    fn color(&self) -> u8 {
        /*使用前景色作为一个基数据，占用低4位
        把背景色左移来占用高位
        然后把这两个位合并到一起
        */
        let fg = self.foreground as u8;
        let bg = (self.background as u8) << 4;
        fg | bg
    }

    /*处于方便，输入数据将会使用一个原始的字节流，
    而没有使用一个能保证编码正确的类型
    */
    fn print(&mut self, text: &[u8]) {
        let color = self.color();

        let framebuffer = 0xb8000 as *mut u8;

        for &character in text {
            unsafe {
                framebuffer.offset(self.position).write_volatile(character);
                framebuffer.offset(self.position + 1).write_volatile(color);
            }
            self.position += 2;
        }
    }
}
////////Cursor的定义和所实现的方法//////////
#[panic_handler]
#[no_mangle]
pub fn panic(_info: &PanicInfo) -> ! {
    unsafe {
        intrinsics::abort();
    }
}

#[lang = "eh_personality"]
#[no_mangle]
pub extern "C" fn eh_personality() {}

/////////输出到屏幕的功能////////////
#[no_mangle]
pub extern "C" fn _start() -> ! {
    let text = b"Rust in Action";

    let mut cursor = Cursor {
        position: 0,
        foreground: Color::BrightCyan,
        background: Color::Black,
    };
    cursor.print(text);

    loop {
        hlt();
    }
}

////////为Cursor类型实现core::fmt::Write/////////
impl fmt::Write for Cursor {
    fn write_str(&mut self, s: &str) -> fmt::Result {
        self.print(s.as_bytes());
        Ok(())
    }
}

#[panic_handler]
#[no_mangle]
pub fn panic(info: &PanicInfo) -> ! {
    let mut cursor = Cursor {
        position: 0,
        foreground: Color::White,
        background: Color::Red,
    };
    for _ in 0..(80 * 25) {
        cursor.print(b" ");
    }
    cursor.position = 0;
    write!(cursor, "{}", info).unwrap();

    loop {
        unsafe {
            hlt();
        }
    }
}
